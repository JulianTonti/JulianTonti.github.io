**Different ways to define functions** 
             function square(num)    { return num*num; }
let square = function square(num)    { return num*num; }
let square = function       (num)    { return num*num; }
let square =                (num) => { return num*num; }
let square =                 num  => { return num*num; }
let square =                (num) =>          num*num;
let square =                 num  =>          num*num;
* braces {} and return can be omitted for lambda functions with only one line. * parens () can be omitted for lambda functions with only one input.
